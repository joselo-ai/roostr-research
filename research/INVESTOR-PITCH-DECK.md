**Example Agent Code:**
```python
class WarrenBuffettAgent:
    def evaluate(self, signal):
        # Analyze moat
        has_moat = self.detect_moat_keywords(signal.catalyst)
        
        # Analyze quality
        roe = signal.fundamentals.get('roe', 0)
        quality_score = roe / 15.0  # ROE 15%+ = quality
        
        # Calculate conviction
        conviction = (has_moat * 5) + quality_score
        
        return {
            "investor": "Warren Buffett",
            "conviction": min(conviction, 10),
            "vote": "BUY" if conviction >= 7 else "HOLD",
            "reasoning": f"ROE {roe}%, moat detected: {has_moat}"
        }
```
